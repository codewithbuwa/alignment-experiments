\section{Data and hyperparameters}

\begin{frame}{The data}
    \begin{block}{DPO dataset}
        \begin{itemize}
            \item Number of samples: 1000
            \item points sampled from \(\mathcal{N}(5, 2^2)\) and \(0.5\mathcal{N}(3, 2^2) + 0.5\mathcal{N}(8, 2^2)\)
            \item winners represent points close to \(7\)
        \end{itemize}
    \end{block}

    \vspace{0.3cm}

    \begin{block}{KTO dataset}
        \begin{enumerate}
            \item Number of samples: 1000
            \item points sampled from \(\mathcal{N}(5, 2^2)\) and \(0.5\mathcal{N}(3, 2^2) + 0.5\mathcal{N}(8, 2^2)\)
            \item positive feedback are values in the interval \([7 - \delta, 7 + \delta]\) (\(\delta = 1.5\))
        \end{enumerate}
    \end{block}

    \begin{block}{Reward}
        \begin{itemize}
            \item Oracle reward: \(-|y-7|\)
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Imbalanced set construction}
\begin{block}{For KTO}
    \begin{itemize}
        \item The sample points are chosen such that a proportion lies in the desirable zone.
        \item \(\alpha\) is the parameter that controls how much positive feedback we need.
    \end{itemize}
\end{block}

\begin{block}{For DPO}
    \begin{itemize}
        \item Points are sampled everywhere except in a neighborhood near \(7\), \(\mathcal{B}(7, (1-\alpha)\sigma)\).
        \item A small value of \(\alpha\) means that we remove most highly rewarded points.
        \item When \(\alpha = 1\), we recover the standard DPO construction.
    \end{itemize}
\end{block}
\end{frame}

\begin{frame}{Hyperparameters}
    \begin{itemize}
        \item To ensure reproducibility and valid comparison, all hyperparameters are fixed across DPO and KTO.
    \end{itemize}

    \begin{table}
        \centering
        \begin{tabular}{cc}
        \hline
            \textbf{Hyperparameters} & \textbf{Values}\\
            \hline
            \(\beta\) & \(1.0\)\\
            \(\lambda\) & \(1.33\)\\
            Optimizer & Adam\\
            Learning rate & \(1\times10^{-3}\)\\
            \hline
        \end{tabular}
        \caption{Training hyperparameters}
    \end{table}
\end{frame}

\section{Imbalanced Experiments}

\begin{frame}{Imbalanced Set Construction Changes What Each Method Learns}
    \begin{itemize}
        \item We compare both methods when useful supervision becomes scarce.
        \item For DPO, imbalance means fewer near-target high-quality preference pairs.
        \item For KTO, imbalance means fewer positive labels inside the desirable zone.
        \item The two methods react differently to this reduction in signal.
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[width=0.46\linewidth]{figures/single_imbalance_density.png}
    \end{figure}
\end{frame}

\begin{frame}{Shared Evaluation: DPO Peaks, KTO Covers}
    \begin{itemize}
        \item After training, both policies are evaluated in the same downstream space.
        \item Mean reward measures closeness to \(7\).
        \item Zone occupancy measures how much mass lies in the acceptable region.
        \item For single Gaussians, \(\sigma\) is enough to describe spread.
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[width=0.58\linewidth]{figures/single_imbalance_shared_eval.png}
    \end{figure}
\end{frame}

\begin{frame}{DPO Under Imbalance: Does Initialization Still Lead Toward 7?}
    \begin{itemize}
        \item We initialize the policy around, left, and right of the target.
        \item We compare full supervision (\(\alpha = 1.0\)) and weak supervision (\(\alpha = 0.1\)).
        \item This tests whether DPO still moves toward the best accessible region.
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[width=0.86\linewidth,height=0.46\textheight,keepaspectratio]{figures/single_dpo_init_imbalance_grid.png}
    \end{figure}
\end{frame}

\begin{frame}{KTO Remains Region-Seeking Across Initializations}
    \begin{itemize}
        \item KTO is tested with the same left, right, and around initializations.
        \item Even with fewer positive samples, it remains broader than DPO.
        \item Its behavior is guided by zone coverage rather than point concentration.
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[width=0.86\linewidth,height=0.46\textheight,keepaspectratio]{figures/single_kto_init_imbalance_grid.png}
    \end{figure}
\end{frame}

\begin{frame}{How DPO and KTO Infer After Training}
    \begin{itemize}
        \item We sample from the reference policy before training, then from the trained DPO and KTO policies.
        \item DPO tends to concentrate near the best accessible value.
        \item KTO tends to sample across a broader acceptable region.
        \item Below is the left-initialization case, where robustness is easiest to see.
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[width=0.92\linewidth]{figures/sample_before_after_left.png}
    \end{figure}
\end{frame}

\begin{frame}{Three Components Make Initialization Effects Visible}
    \begin{itemize}
        \item We repeat the initialization analysis with a 3-component Gaussian mixture.
        \item Left, right, and around initializations are tracked component by component.
        \item Means, sigmas, and weights reveal which components absorb the initialization bias.
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[width=0.86\linewidth]{figures/mix_init_componentwise_3mix.png}
    \end{figure}
\end{frame}

\begin{frame}{Takeaway From Imbalanced Training}
    \begin{itemize}
        \item DPO and KTO remain qualitatively different under weak supervision.
        \item DPO is sharper and more point-seeking.
        \item KTO is broader and more zone-seeking.
        \item Initialization checks show whether this behavior remains directional and robust.
        \item Mixtures reveal how that behavior is distributed across components.
    \end{itemize}
\end{frame}
